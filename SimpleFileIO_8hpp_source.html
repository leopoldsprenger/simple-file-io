105 inline std::string formatIOError(IOError code,
106 const std::string& path = "",
107 const std::string& detail = "") {
108 switch (code) {
109 case IOError::FileNotOpen:
110 return "File operation failed 'file is not open': " + path;
111 case IOError::FileNotFound:
112 return "File not found: " + path;
113 case IOError::PermissionDenied:
114 return "Permission denied while accessing: " + path;
115 case IOError::ReadError:
116 return "Low-level read error" +
117 (detail.empty() ? "" : (": " + detail));
118 case IOError::WriteError:
119 return "Low-level write error" +
120 (detail.empty() ? "" : (": " + detail));
121 default:
122 return "Unknown I/O error.";
123 }

256 inline std::string TextReader::readLine() {
257 if (!file)
258 throw IOException(IOError::FileNotOpen, formatIOError(IOError::FileNotOpen, path), path);
259
260 std::string line;
261 bool anyDataRead = false;
262
263 while (true) {
264 if (cursor >= bufferEnd) {
265 // refill buffer
266 size_t leftover = bufferEnd - cursor;
267 if (leftover > 0) {
268 std::memmove(buffer.data(), buffer.data() + cursor, leftover);
269 }
270 size_t bytesRead = SFIO_FREAD(buffer.data() + leftover, 1, buffer.size() - leftover, file);
271 cursor = 0;
272 bufferEnd = leftover + bytesRead;
273
274 if (bytesRead == 0) {
275 if (ferror(file))
276 throw IOException(IOError::ReadError, formatIOError(IOError::ReadError, path), path);
277 break; // normal EOF
278 }
279 }
280
281 size_t start = cursor;
282 while (cursor < bufferEnd && buffer[cursor] != '\n') cursor++;
283 if (cursor > start) {
284 line.append(buffer.data() + start, cursor - start);
285 anyDataRead = true;
286 }
287
288 if (cursor < bufferEnd && buffer[cursor] == '\n') {
289 cursor++; // skip newline
290 break;
291 }
292 }
293
294 if (!anyDataRead && std::feof(file))
295 return {}; // EOF reached, return empty string without throwing
296
297 return line;
298 }

300 inline std::vector<std::string> TextReader::readLines(int numLines) {
301 std::vector<std::string> lines;
302 if (numLines > 0) lines.reserve(numLines);
303
304 while (numLines == 0 || lines.size() < static_cast<size_t>(numLines)) {
305 std::string line = readLine();

307 lines.push_back(std::move(line));

407 inline void TextWriter::writeString(const std::string& data) {
408 // chunked write to avoid issues with extremely large strings
409 const size_t chunkSize = 1 << 20;
410 size_t offset = 0;
411 while (offset < data.size()) {

433 inline void TextWriter::writeLines(const std::vector<std::string>& lines) {
434 if (!file)
435 throw IOException(IOError::FileNotOpen, formatIOError(IOError::FileNotOpen, path), path);
436 if (lines.empty()) return;
437
438 // Compute total size: sum of line sizes + missing newlines for lines that don't end with '\n'
439 size_t totalSize = 0;
440 size_t missingNewlines = 0;
441 for (const auto &line : lines) {
442 totalSize += line.size();
443 if (line.empty() || line.back() != '\n') ++missingNewlines;
444 }
445 totalSize += missingNewlines;
446
447 // Allocate once and memcpy all data into the buffer to avoid per-line overhead
448 buffer.clear();
449 buffer.resize(totalSize);
450 char* ptr = buffer.data();
451
452 for (const auto &line : lines) {
453 size_t len = line.size();
454 if (len > 0) {
455 std::memcpy(ptr, line.data(), len);
456 ptr += len;
457 }
458 if (len == 0 || line[len-1] != '\n') {
459 *ptr++ = '\n';
460 }
461 }
462
463 size_t written = SFIO_FWRITE(buffer.data(), 1, buffer.size(), file);
464 if (written != buffer.size())
465 throw IOException(IOError::WriteError, formatIOError(IOError::WriteError, path, "Failed to write lines to file."), path);
466 }
